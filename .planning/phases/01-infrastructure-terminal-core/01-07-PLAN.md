    **backend/app/services/claude.py** — per RESEARCH.md detection pattern:
    ```python
    import asyncssh

    async def detect_claude_sessions(conn: asyncssh.SSHClientConnection) -> list[dict]:
        """Detect running Claude Code sessions on a remote machine."""
        try:
            result = await conn.run(
                "tmux list-windows -a -F '#{session_name}:#{window_index}:#{window_name}:#{pane_current_command}' 2>/dev/null",
                check=True,
            )
            sessions = []
            for line in result.stdout.strip().split("\n"):
                if not line:
                    continue
                parts = line.split(":")
                if len(parts) >= 4 and "claude" in parts[3].lower():
                    sessions.append({
                        "tmux_session": parts[0],
                        "window_index": int(parts[1]),
                        "window_name": parts[2],
                        "command": parts[3],
                        "status": "running",  # default, refined by output monitoring
                    })
            return sessions
        except asyncssh.ProcessError:
            return []
    ```

    Add `detect_waiting_for_input` function — monitor terminal output for Claude Code prompt patterns:
    ```python
    async def detect_waiting_for_input(conn: asyncssh.SSHClientConnection, tmux_session: str, window_index: int) -> bool:
        """Check if a Claude Code session is waiting for user input."""
        try:
            result = await conn.run(
                f"tmux capture-pane -t {tmux_session}:{window_index} -p | tail -5",
                check=True,
            )
            output = result.stdout.strip()
            # Claude Code shows prompts like "> " or waiting indicators
            waiting_patterns = ["> ", "? ", "waiting for", "approve", "deny"]
            return any(pattern in output.lower() for pattern in waiting_patterns)
        except asyncssh.ProcessError:
            return False
    ```
