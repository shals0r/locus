    **frontend/src/stores/machineStore.ts** — Zustand store:
    ```typescript
    import { create } from "zustand";
    import { Machine, MachineStatus } from "../types";

    interface MachineState {
      machines: Machine[];
      activeMachineId: string | null;
      machineStatuses: Record<string, MachineStatus>;
      setMachines: (machines: Machine[]) => void;
      addMachine: (machine: Machine) => void;
      removeMachine: (id: string) => void;
      setActiveMachine: (id: string) => void;
      setMachineStatus: (id: string, status: MachineStatus) => void;
    }

    export const useMachineStore = create<MachineState>((set) => ({
      machines: [],
      activeMachineId: null,
      machineStatuses: {},
      setMachines: (machines) => set({ machines }),
      addMachine: (machine) => set((s) => ({ machines: [...s.machines, machine] })),
      removeMachine: (id) => set((s) => ({
        machines: s.machines.filter((m) => m.id !== id),
        activeMachineId: s.activeMachineId === id ? null : s.activeMachineId,
      })),
      setActiveMachine: (id) => set({ activeMachineId: id }),
      setMachineStatus: (id, status) => set((s) => ({
        machineStatuses: { ...s.machineStatuses, [id]: status },
      })),
    }));
    ```

    **frontend/src/stores/sessionStore.ts** — Zustand store:
    ```typescript
    import { create } from "zustand";
    import { TerminalSession } from "../types";

    interface SessionState {
      sessions: TerminalSession[];
      activeSessionId: string | null;
      setSessions: (sessions: TerminalSession[]) => void;
      addSession: (session: TerminalSession) => void;
      removeSession: (id: string) => void;
      setActiveSession: (id: string) => void;
      getSessionsForMachine: (machineId: string) => TerminalSession[];
    }

    export const useSessionStore = create<SessionState>((set, get) => ({
      sessions: [],
      activeSessionId: null,
      setSessions: (sessions) => set({ sessions }),
      addSession: (session) => set((s) => ({ sessions: [...s.sessions, session] })),
      removeSession: (id) => set((s) => ({
        sessions: s.sessions.filter((sess) => sess.id !== id),
        activeSessionId: s.activeSessionId === id ? null : s.activeSessionId,
      })),
      setActiveSession: (id) => set({ activeSessionId: id }),
      getSessionsForMachine: (machineId) => get().sessions.filter((s) => s.machine_id === machineId),
    }));
    ```
