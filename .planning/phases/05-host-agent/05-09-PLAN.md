Add a module-level async helper function (outside the Settings class):
```python
async def get_local_scan_paths_from_db() -> list[str] | None:
    """Read local_repo_scan_paths from DB. Returns None if not set."""
    from app.database import async_session_factory
    from app.models.app_setting import AppSetting
    try:
        async with async_session_factory() as db:
            result = await db.get(AppSetting, "local_repo_scan_paths")
            if result and result.value:
                return [p.strip() for p in result.value.split(",") if p.strip()]
    except Exception:
        pass
    return None
```

**Create frontend/src/components/settings/GeneralSettings.tsx:**

Build a settings form component for configuring local repo scan paths:

```tsx
import { useState, useEffect } from "react";
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { FolderOpen, Plus, X, Save, Loader2 } from "lucide-react";
import { apiGet, apiPut } from "../../hooks/useApi";

interface GeneralSettingsData {
  local_repo_scan_paths: string[];
}

export function GeneralSettings() {
  // State
  const [paths, setPaths] = useState<string[]>([]);
  const [newPath, setNewPath] = useState("");
  const [dirty, setDirty] = useState(false);
  const queryClient = useQueryClient();

  // Fetch current settings
  const { data, isLoading } = useQuery({
    queryKey: ["settings", "general"],
    queryFn: () => apiGet<GeneralSettingsData>("/api/settings/general"),
  });

  // Sync state when data loads
  useEffect(() => {
    if (data) {
      setPaths(data.local_repo_scan_paths);
      setDirty(false);
    }
  }, [data]);

  // Save mutation
  const saveMutation = useMutation({
    mutationFn: (newPaths: string[]) =>
      apiPut<GeneralSettingsData>("/api/settings/general", {
        local_repo_scan_paths: newPaths,
      }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["settings", "general"] });
      setDirty(false);
    },
  });

  const addPath = () => {
    const trimmed = newPath.trim();
    if (trimmed && !paths.includes(trimmed)) {
      setPaths([...paths, trimmed]);
      setNewPath("");
      setDirty(true);
    }
  };

  const removePath = (index: number) => {
    setPaths(paths.filter((_, i) => i !== index));
    setDirty(true);
  };

  const handleSave = () => {
    saveMutation.mutate(paths);
  };

  // Render (follow CredentialSettings/MachineSettings pattern: Tailwind dark theme, compact)
  // - Show loading spinner while fetching
  // - List of configured paths with X button to remove each
  // - Input field + Add button to add new path
  // - Hint text: "Absolute paths to directories containing git repos (e.g., C:\\Users\\you\\projects or /home/you/projects)"
  // - Save button appears when dirty, disabled during mutation
  // - Success toast or inline "Saved" text on successful save
}
```
