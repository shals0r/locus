Convert each of the 4 existing adapter classes into standalone Python scripts following the D-02 worker contract. Each script must:

1. Be a standalone executable (`#!/usr/bin/env python3`)
2. Read configuration from environment variables (injected by supervisor):
   - `LOCUS_INGEST_URL` -- URL to POST results to (e.g., `http://localhost:8080/api/feed/ingest`)
   - `LOCUS_WORKER_SECRET` -- HMAC secret for auth with ingest API
   - `POLL_INTERVAL` -- seconds between polls
   - Service-specific credential env vars (e.g., `GITHUB_TOKEN`, `GITLAB_TOKEN`, `JIRA_TOKEN`, etc.)
   - Service-specific config env vars prefixed with `WORKER_` (e.g., `WORKER_REPOS`, `WORKER_BASE_URL`, `WORKER_PROJECT_ID`)
3. Define a `poll() -> list[dict]` function that returns items matching the IngestPayload schema
4. Have a `__main__` loop that: calls `poll()`, POSTs each item to `LOCUS_INGEST_URL` with `X-Locus-Worker-Secret` header, sleeps for `POLL_INTERVAL`, catches all exceptions and prints errors to stdout (supervisor reads stdout)
5. Use `httpx` for HTTP requests (already installed in Docker image)
6. Print structured log lines to stdout: `{ISO_TIMESTAMP} {LEVEL} {message}` format (e.g., `2026-03-28T14:32:01Z INFO Polled 12 items from GitHub`)
7. Use `flush=True` on all print statements to avoid stdout buffering issues (research Pitfall 1)

async def list_workers(db: AsyncSession) -> list[IntegrationSource]:
    """List all integration sources ordered by name."""
    result = await db.execute(select(IntegrationSource).order_by(IntegrationSource.name))
    return list(result.scalars().all())

async def get_worker(db: AsyncSession, worker_id: str) -> IntegrationSource | None:
    result = await db.execute(select(IntegrationSource).where(IntegrationSource.id == worker_id))
    return result.scalar_one_or_none()

async def create_worker(db: AsyncSession, data: dict) -> IntegrationSource:
    """Create a new worker entry. Sets worker_script_path based on naming convention."""
    source = IntegrationSource(**data)
    db.add(source)
    await db.flush()
    return source

**Worker REST API** (`backend/app/api/workers.py`):

Follow existing router patterns (see `api/machines.py`, `api/tasks.py`):

```python
router = APIRouter(prefix="/api/workers", tags=["workers"])

@router.get("", response_model=WorkerListResponse)
async def list_workers(...)  # List all workers with status

@router.get("/{worker_id}", response_model=WorkerResponse)
async def get_worker(...)  # Get single worker

@router.post("", response_model=WorkerResponse, status_code=201)
async def create_worker(...)  # Create new worker entry

@router.patch("/{worker_id}", response_model=WorkerResponse)
async def update_worker(...)  # Update worker config (per D-17 quick config)

@router.delete("/{worker_id}")
async def delete_worker(...)  # Delete worker (must be stopped first)

@router.post("/{worker_id}/start", response_model=WorkerActionResponse)
async def start_worker(...)  # Start worker subprocess via supervisor

@router.post("/{worker_id}/stop", response_model=WorkerActionResponse)
async def stop_worker(...)  # Stop worker subprocess via supervisor

@router.post("/{worker_id}/restart", response_model=WorkerActionResponse)
async def restart_worker(...)  # Restart worker

@router.post("/{worker_id}/enable", response_model=WorkerActionResponse)
async def enable_worker(...)  # Re-enable disabled worker (D-09): reset failure_count, set status to stopped

@router.get("/{worker_id}/logs")
async def get_worker_logs(...)  # Return buffered log lines as JSON array
```

All endpoints require `_user: dict = Depends(get_current_user)` for JWT auth.

The `start_worker` endpoint: looks up IntegrationSource, calls `supervisor.build_worker_env()`, calls `supervisor.start_worker()`, updates DB status.
The `stop_worker` endpoint: calls `supervisor.stop_worker()`, updates DB status to "stopped".
The `enable_worker` endpoint: resets `failure_count` to 0, sets `worker_status` to "stopped" (user must click start separately, or auto-start on enable -- use auto-start per D-09).
